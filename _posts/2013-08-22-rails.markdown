四个模型 category, tag, list, product
model
class Categroy < ActiveRecord::Base
has_many :tags
end

class Tag < ActiveRecord::Base
has_many :lists
belongs_to :category
end

class List < ActiveRecord::Base
belongs_to :tag
has_many :products
end

class Product < ActiveRecord::Base
belongs_to :list
end
